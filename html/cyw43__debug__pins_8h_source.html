43 static inline void logic_debug_init(void) {
44  #if CYW43_LOGIC_DEBUG
45  const int dbg_start = 2;
46  const int dbg_end = 14;
47  for (int pin = dbg_start; pin <= dbg_end; pin++) {
48  gpio_init(pin);
49  gpio_set_dir(pin, 1);
50  gpio_put(pin, 0);
51  }
52  #endif
53 }
54 
55 static inline void logic_debug_set(uint8_t pin, uint8_t value) {
56  #if CYW43_LOGIC_DEBUG
57  gpio_put(pin, value);
58  #else
59  (void)pin;
60  (void)value;
61  #endif
62 }
