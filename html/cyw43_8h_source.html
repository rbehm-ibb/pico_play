533 static inline uint32_t cyw43_pm_value(uint8_t pm_mode, uint16_t pm2_sleep_ret_ms, uint8_t li_beacon_period, uint8_t li_dtim_period, uint8_t li_assoc) {
534  return li_assoc << 20 | // listen interval sent to ap
535  li_dtim_period << 16 |
536  li_beacon_period << 12 |
537  (pm2_sleep_ret_ms / 10) << 4 | // cyw43_ll_wifi_pm multiplies this by 10
538  pm_mode; // CYW43_PM2_POWERSAVE_MODE etc
539 }
